export access_token=$(\
curl -X POST http://localhost:8080/realms/hfs-realm/protocol/openid-connect/token \
-H 'content-type: application/x-www-form-urlencoded' \
-d 'client_id=hfs-frontend' \
-d 'username=admin&password=admin&grant_type=password' | jq --raw-output '.access_token' \
)

curl http://localhost:8000/admParameterCategory/1 \
-H "Authorization: Bearer "$access_token

curl http://localhost:8000/funcionario?nome=HENRIQUE FIGUEIREDO \
-H "Authorization: Bearer "$access_token


mvnw package -f pom.xml
